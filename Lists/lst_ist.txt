# Fixed: the original ismy_insert created an empty list instead of using the
# "a" argument, so the loop over range(len(lst)) never ran and it always
# returned None. This version inserts num into a copy of a in sorted position.
def ismy_insert(a, num):
    lst = a[:]
    for i in range(len(lst)):
        if num <= lst[i]:
            return lst[:i] + [num] + lst[i:]
    lst.append(num)
    return lst

def my_pop(lst):
    num = lst[-1]
    lst = lst[:len(lst) - 1]
    return num, lst

# Fixed: the original used lst[:] - lst[i], which is not valid for Python lists,
# and referenced an undefined variable "i". This removes the element at index n.
def my_del(lst, n):
    num = lst[n]
    new_lst = lst[:n] + lst[n + 1:]
    return num, new_lst
