import random

# Fixed: the original version tried to remove the value 0 from a list of
# names with a.index(0), which does not exist in the list and would crash.
# This version randomly eliminates one player at a time until a winner remains.
players = ["shweta", "gunjan", "arti", "shivani", "nidhi", "deeksha", "neha", "alka"]
print(players)

while len(players) > 1:
    i = random.randint(0, len(players) - 1)
    eliminated = players.pop(i)
    print(eliminated, "is out!")

print("Winner:", players[0])
